def add(severity, message = nil, progname = nil)
  severity ||= UNKNOWN
  @logdev = logdevice(severity)

  if @logdev.nil? || severity < @level
    return true
  end
  progname ||= @progname
  if message.nil?
    if block_given?
      message = yield
    else
      message = progname
      progname = @progname
    end
  end
  @logdev.puts(
    format_message(format_severity(severity), Time.now, progname, message)
  )
  true
end

def logdevice(severity)
  if severity > INFO
    $stderr
  else
    $stdout
  end
end
